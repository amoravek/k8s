Taints, tolerations

kubectl taint nodes node1 key1=value1:NoSchedule-

 1 tolerations:
 2 - key: "key1"
 3   operator: "Equal"
 4   value: "value1"
 5   effect: "NoSchedule"
 6 
 7 !yaml
 8 tolerations:
 9 - key: "key1"
10   operator: "Exists"
11   effect: "NoSchedule"
